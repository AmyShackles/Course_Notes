#### Decimal to Binary

int decimalToBinary(int number, char digitChar) {

#### Binary to Decimal

while (digitChar != 10) {

  }
  return number;

#### Hexadecimal to Decimal

```
int hexadecimalToDecimal(int number, char digitChar) {
  if (number > 9) {
    number = number - 7;
  }
  while (digitChar !=10) {
    if ((digitChar - '0') > 9) {
      number = number * 16 + ((digitChar - '0') - 7);
    } else {
      number = number * 16 + (digitChar - '0');
    }
    digitChar = cin.get();
  }
  return number;
}
```
